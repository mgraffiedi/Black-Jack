int gotoxy(int x, int y) { 
    //setto le nuove coordinate del cursore... 
    ConsoleInfo.dwCursorPosition.Y = y; 
    ConsoleInfo.dwCursorPosition.X = x; 

    //chiamo la funzione che aggiorna il buffer della console, con i nuovi attributi 
    SetConsoleCursorPosition(Hout,ConsoleInfo.dwCursorPosition);
	return 0;
} 

void textcolor(unsigned short color) {
     HANDLE hCon = GetStdHandle(STD_OUTPUT_HANDLE);
     SetConsoleTextAttribute(hCon,color);
}

void creamazzo(int nmazzi){
     int i,j,g;
     for (i=0;i<nmazzi;i++){

             for (g=1;g<14;g++){
                 v[i*52+((j-1)*13)+(g-1)]=j*100+g;                                
             }  
         }                       
                           
      }     
     
}
void sort(int n,int nmazzi){
     int i,ni,c,p,pp;
     ni=52*nmazzi;
     n=n*nmazzi*20;
     for (i=0;i<n;i++){
         p=rand()%ni;      
         pp=rand()%ni;         
         c=v[p];
         v[p]=v[pp];
         v[pp]=c;       
     }
     
}

int scrivicarta(int card,int x,int y){

     switch (c){
            case 1:case 2:textcolor(44); break;
            default:textcolor(32); break;
            
            }
     switch (card){
            case 1:
                 cout<<"A";
                 break;  
            case 11:
                 cout<<"J";
                 break; 
            case 12:
                 cout<<"Q";
                 break; 
            case 13:
                 cout<<"K";
                 break;        
            default:
                    cout<<card;
                    break;            
                  
     }
     s=c+2;
     if(card!=10){cout<<" ";}
     cout<<s; 
     textcolor(47);
     if ((card>9)||(card==1)){

}

int scelta(int x,int y,int max){
    int s,ch;
    s=0;
    do{
       gotoxy(x+1,y+s);
       textcolor(4);
       cout<<"+";    
       gotoxy(x+1,y+max+1);
       ch=getch();
       gotoxy(x+1,y+s);cout<<" ";
       switch (ch){
              case 72:  
                   if(s>0){s--;}
                   break;  
              case 80:  
                   if(s<max){s++;}
                   break;      
                  
                  
       }
    }while(ch!=13);
    textcolor(15);
    return s;
}

int pescata(int carta,int &A,int punti){
    int y;
    y=carta/100;
    y=carta-(y*100);
    if(y>10){y=10;}
    if (A!=0){
       if (punti+y>21){
          punti=punti+y-10;
          A=0;
       }else{
             punti+=y; 
             }
    }else{
          if (y==1){
             if (punti+11>21){
                punti=punti+y;
                A=0;
             }else{
                   punti+=11; 
                   A=1;
             }
                
             
          }else{
             punti+=y;
          }
    }
    return punti;
}
int vincitore(int giocatore,int banco){
    int v;
    v=0;
    if (banco>21){
       if (giocatore>21){
          v=0;
       }else{
             v=1;
       }
    }else{
       if (giocatore>21){
          v=-1;
                        
       }else{
             if (giocatore>banco){
                v=1;                    
             }else{
             if (giocatore<banco){
                v=-1;
             }else{
                v=0;                        
             }
             
             }
             
       }
          
    }
    return v;
    
}
